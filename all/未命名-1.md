```mermaid
stateDiagram
    [*] --> INIT : 系统启动
    state INIT {
        :初始化状态;
    }
    
    INIT --> WAIT_ANSWER : 出题
    state WAIT_ANSWER {
        :等待用户输入;
    }
    
    WAIT_ANSWER --> ANSWER_CORRECT1 : 答对
    WAIT_ANSWER --> ANSWER_WRONG : 答错或超时
    
    state ANSWER_CORRECT1 {
        :答对第一题;
    }
    
    ANSWER_CORRECT1 --> ANSWER_CORRECT2 : 答对第二题
    
    state ANSWER_CORRECT2 {
        :答对第二题;
    }
    
    ANSWER_CORRECT2 --> OUTPUT_PULSE : 输出脉冲
    
    OUTPUT_PULSE --> WAIT_ANSWER : 继续下一题
    
    state ANSWER_WRONG {
        :用户答错;
    }
    
    ANSWER_WRONG --> WAIT_ANSWER : 重新计数

```
